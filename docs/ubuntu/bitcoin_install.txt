1. openssl 설치
	- cd /home/path
	- mkdir download
	- cd download
	- wget https://www.openssl.org/source/openssl-1.0.1l.tar.gz
	- tar -xzvf openssl-1.0.1l.tar.gz
	- cd openssl-1.0.1l
	- ./config --prefix=/usr/local shared enable-ec enable-ecdh enable-ecdsa // root 계정으로 실행
	- make // root 계정으로 실행
	- make install // root 계정으로 실행

2. Berkeley DB4.8 설치
	- add-apt-repository ppa:bitcoin/bitcoin
	- apt-get update
	- apt-get install libdb4.8-dev libdb4.8++-dev

3. bitcoin 설치
	- mkdir ~/bitcoin
	- cd ~/download
	- git clone https://github.com/bitcoin/bitcoin.git
	- cd bitcoin
	- patch -p0 < bitcoin_brainboot.patch
	- git diff
	- ./autogen.sh // root 계정으로 실행
	- ./configure --prefix=/home/path/bitcoin --with-gui=no --without-qt // root 계정으로 실행
	- make // root 계정으로 실행
	- make install // root 계정으로 실행

4. bitcoin conf 작성
	- mkdir .bitcoin
	- cd ~/.bitcoin
	- vim bitcoin.conf

		#testnet=1
		txindex=1
		upnp=1
		listen=1
		server=1
		daemon=0
		dbcache=16000
		maxmempool=200
		maxreceiverbuffer=2500
		maxsendbuffer=500
		deprecatedrpc=signrawtransaction

		rpcuser=userid
		rpcpassword=password

5. bitcoin 실행
	- /path/bitcoin/bin/bitcoind --daemon
